    let mergeSort = (arr) => { 
        let len = arr.length,left=[],right=[]; 
        if(len<2){ 
            return arr; 
        } 
        for(i=1;i<len;i++){ 
            if(arr[i]<=arr[0]){ 
                left.push(arr[i]) 
            }else{ 
                right.push(arr[i]); 
            } 
        } 
        return mergeSort(left).concat(arr[0],mergeSort(right)); 
    }

    let newList = [];
    function distinct(arr) {
        for (let item of arr) {
            let { serial } = item;
            let ser = newList.find(item => item.serial === serial);
            if (!ser) {
                newList.push(item);
            }
        }
        return newList;
    }
